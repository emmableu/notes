import torch  
filename = "../../makemore/names.txt"  
def name_to_bigram(filename):  
    names = open(filename).read().splitlines()  
    N = torch.zeros(27, 27)  
    itos = '.abcdefghijklmnopqrstuvwxyz'  
    stoi = {ele:i for i, ele in enumerate(itos)}  
    for name in names:  
        ele = ['.'] + [c for c in name] + ['.']  
        for ch1, ch2 in zip(ele, ele[1:]):  
            N[stoi[ch1], stoi[ch2]] += 1  
    return N  
  
N = name_to_bigram(filename)  
print(N.to(dtype=torch.int32))

import torch
def build_bigram_probability_matrix(N: torch.tensor) -> torch.tensor:
    prob_matrix = N
    sum = N.sum(dim=1, keepdim=True) # [2,3] => [2, 1] * must have keepdim *
    prob_matrix = prob_matrix/sum
    return prob_matrix

import torch
from typing import List
def generate_words(prob_matrix: torch.tensor, itos: str, num_words: int) -> List[str]:
    def dfs(i, cur_path, prob_matrix):
        if i == 0 and len(cur_path) > 0:
            return cur_path
        dist = torch.distributions.Categorical(prob_matrix[i])
        next_i = dist.sample((1,)).item()
        return dfs(next_i, cur_path + [next_i], prob_matrix)

    out = []
    for _ in range(num_words):
        path = dfs(0, [], prob_matrix)
        word = "".join([itos[ele] for ele in path])
        out.append(word)
    return out

def prepare_bigram_data(names: List[str]) -> Tuple[torch.Tensor, torch.Tensor]:  
    x_data = []  
    y_data = []  
    for name in names:  
        ele = '.' + name + '.'  
        for c1, c2 in zip(ele, ele[1:]):  
            x_data.append(stoi[c1])  
            y_data.append(stoi[c2])  
    X = F.one_hot(torch.tensor(x_data), num_classes=len(itos))  
    y = F.one_hot(torch.tensor(y_data), num_classes=len(itos))  
    return X, y

import torch
from typing import List, Tuple
def train_test_split(X: torch.Tensor, y: torch.Tensor, ratio: float = 0.9) -> Tuple[torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor]:
    size = X.size(0)
    idx = torch.randperm(size)
    X, y = X[idx], y[idx]
    split = int(size*ratio)
    X_train = X[:split, :]
    y_train = y[:split, :]
    X_test = X[split:, :]
    y_test = y[split:, :]
    return X_train, y_train, X_test, y_test

import evaluate

def calc_metrics(y_pred, y_test):
    accuracy_metrics = evaluate.load('accuracy')
    precision_metrics = evaluate.load('precision')
    recall_metrics = evaluate.load('recall')
    f1_metrics = evaluate.load('f1')

    accuracy = accuracy_metrics.compute(predictions=y_pred, references=y_test)
    precision = precision_metrics.compute(predictions=y_pred, references=y_test, average='macro')
    recall = recall_metrics.compute(predictions=y_pred, references=y_test, average='macro')
    f1 = f1_metrics.compute(predictions=y_pred, references=y_test, average='macro')
    return accuracy['accuracy'], precision['precision'], recall['recall'], f1['f1']

def calc_loss(logits, y_test):
    loss_fn = nn.CrossEntropyLoss()
    loss = loss_fn(input=logits, target=y_test)
    return loss


def train_model(model, X_train, y_train, X_test, y_test):
    loss_lst = []
    train_metrics_lst = []
    test_metrics_lst = []
    optimizer = torch.optim.AdamW(model.parameters(), lr=0.01)
    for i in range(1000):
        logits = model(X_train)
        optimizer.zero_grad()
        loss = calc_loss(logits, y_train)
        loss.backward()
        optimizer.step()
        loss_lst.append(loss.item())
        if i%100 == 0:
            y_pred_train = torch.argmax(logits, dim=-1)
            train_metrics = calc_metrics(y_pred_train, y_train)
            train_metrics_lst.append(train_metrics)

            test_logits = model(X_test)
            y_pred_test = torch.argmax(test_logits, dim=-1)
            test_metrics = calc_metrics(y_pred_test, y_test)
            test_metrics_lst.append(test_metrics)
            print(f'iteration --------------- {i}')
            print(f'{train_metrics=}')
            print(f'{test_metrics=}')

    return model, loss_lst, train_metrics_lst, test_metrics_lst

def generate_name(model, cur_i_lst):
    if cur_i_lst[-1] == 0:
        return "".join(itos[ele] for ele in cur_i_lst)
    prev = F.one_hot(torch.tensor(cur_i_lst[-1]), num_classes=27).float()
    logits = model(prev)
    probs = F.softmax(logits)
    dist = torch.distributions.Categorical(probs=probs)
    sample = dist.sample((1,)).item()
    cur_i_lst.append(sample)
    return generate_name(model, cur_i_lst)
