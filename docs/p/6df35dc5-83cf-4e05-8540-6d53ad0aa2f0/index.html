"""
state space: how many points we have now, s
action space: take one number
probability: 1/maxPts
reward: 0 if fail in the end, 1 if not fail in the end
gamma: 1

n = 21, k = 17, maxPts = 10
prob(s=21) = 1 # win
prob(s=20) = 1 # win
prob(s=19) = 1 # win
...
prob(s=16) = 1/10 * prob(s=17) + .. 1/10 * prob(s=21)
prob(s=15) = 1/10 * prob(s=16) + ... + 1/10*prob(s=21)
"""


class Solution:
    def new21Game(self, n: int, k: int, maxPts: int) -> float:
        cur_prob = self.prob(0, n, k, maxPts)
        return cur_prob

    @functools.lru_cache(maxsize=None)
    def prob(self, s, n, k, maxPts):
        if s > n:
            return 0
        if k <= s <= n:
            return 1
        cur_prob = 0
        for i in range(1, maxPts + 1):
            cur_prob += self.prob(s + i, n, k, maxPts) / maxPts

        return cur_prob
