import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

gamma = 0.9
grid_size = 5
A, A_ = (0, 1), (4, 1)
B, B_ = (0, 3), (2, 3)
actions = ['up', 'down', 'left', 'right']

def step(i, j, action):
    if (i, j) == A:
        return A_, 10
    if (i, j) == B:
        return B_, 5

    moves = {'up': (-1, 0), 'down': (1, 0), 'left': (0, -1), 'right': (0, 1)}
    di, dj = moves[action]
    ni, nj = i + di, j + dj

    if 0 <= ni < grid_size and 0 <= nj < grid_size:
        return (ni, nj), 0
    else:
        return (i, j), -1

v = np.zeros((grid_size, grid_size))

for k in range(1000):
    new_v = np.zeros_like(v)
    for i in range(grid_size):
        for j in range(grid_size):
            val = 0
            for a in actions:
                (ni, nj), r = step(i, j, a)
                val += 0.25 * (r + gamma * v[ni, nj])
            new_v[i, j] = val
    if np.max(np.abs(v - new_v)) < 1e-4:
        break
    v = new_v

sns.heatmap(v, annot=True, fmt=".1f", cmap="coolwarm")
plt.title('Policy Evaluation Value Function (Gridworld)')
plt.show()

def policy_evaluation(policy, gamma=0.9, theta=1e-4):
    v = np.zeros((grid_size, grid_size))
    while True:
        delta = 0
        for i in range(grid_size):
            for j in range(grid_size):
                v_old = v[i, j]
                val = 0
                for a in actions:
                    (ni, nj), r = step(i, j, a)
                    val += policy[i, j, a] * (r + gamma * v[ni, nj])
                v[i, j] = val
                delta = max(delta, abs(v_old - val))
        if delta < theta:
            break
    return v

def policy_improvement(v, gamma=0.9):
    policy = np.zeros((grid_size, grid_size, len(actions)))
    stable = True
    for i in range(grid_size):
        for j in range(grid_size):
            q_vals = []
            for a in actions:
                (ni, nj), r = step(i, j, a)
                q_vals.append(r + gamma * v[ni, nj])
            best_a = np.argmax(q_vals)
            for a in range(len(actions)):
                policy[i, j, a] = 1 if a == best_a else 0
    return policy

policy = np.ones((grid_size, grid_size, len(actions))) / len(actions)
for _ in range(10):
    v = policy_evaluation(policy)
    policy = policy_improvement(v)

plt.figure(figsize=(6,5))
sns.heatmap(v, annot=True, fmt=".2f", cmap="YlGnBu")
plt.title('Policy Iteration Final Value Function')
plt.show()

def minPathSum(grid):
    m, n = len(grid), len(grid[0])
    dp = [[0]*n for _ in range(m)]
    dp[-1][-1] = grid[-1][-1]
    for i in reversed(range(m)):
        for j in reversed(range(n)):
            if i == m-1 and j == n-1:
                continue
            down = dp[i+1][j] if i+1 < m else float('inf')
            right = dp[i][j+1] if j+1 < n else float('inf')
            dp[i][j] = grid[i][j] + min(down, right)
    return dp[0][0]
